:Remove parentheses

[tex]81 - \frac{86}{2} + 9 \times 2 - 50 = n[/tex]

:Simplify

[tex]81 - 43 + 9 \times 2 - 50 = n[/tex]

:Simplify 9x2 to 18

[tex]81 - 43 + 18 - 50 = n[/tex]

:Simplify 81-43 to 38

[tex]38 + 18 - 50 = n[/tex]

:Simplify 38+18 to 56

[tex]56 - 50 = n[/tex]

:Simplify 56-50 to 6

[tex]6 = n[/tex]
